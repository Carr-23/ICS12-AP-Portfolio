How To Code It!

// Creates a Class

public class ClassName{

}

//Creating a class constructor (inside class)

public ClassName(){

}

//Creating a overloaded class constructor (inside class)

public ClassName(VariableType VariableName){

}

//Creating Accesor Method (inside class)

public VariableReturnType getVariableName(){

return VariableName;

}

//Creating Mutator Method (inside class)

public setVariableName(VariableType VariableName1){

VariableName = VariableName1;

}

// Creates a SubClass

public class NewClassName extends ClassName{

}

 

 

// Creating a ClassName Object from another class

ClassName cn = new ClassName();

// Creating an overloaded ClassName Object from another class

ClassName cn1 = new ClassName(Value of VariableName);

// Calling an accesor method from another class

System.out.print (cn1.getVariableName());

// Calling an mutator method from another class

cn1.setVariableName(Value of VariableName1)
